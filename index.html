In competitive programming, efficient input and output handling is crucial. Below is a concise and optimized approach for handling input and output in both C++ and Python.

1. Single Integer Input

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    cout << n << "\n";
    
    return 0;
}

Python

import sys

n = int(sys.stdin.readline())
sys.stdout.write(str(n) + "\n")

2. Multiple Space-Separated Inputs

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int a, b;
    cin >> a >> b;
    cout << a + b << "\n";
    
    return 0;
}

Python

import sys

a, b = map(int, sys.stdin.readline().split())
sys.stdout.write(str(a + b) + "\n")

3. Array Input (Space-Separated)

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << "\n";
    
    return 0;
}

Python

import sys

n = int(sys.stdin.readline())
arr = list(map(int, sys.stdin.readline().split()))
sys.stdout.write(" ".join(map(str, arr)) + "\n")

4. Multiple Line Inputs

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < n; i++) {
        cout << arr[i] << "\n";
    }
    
    return 0;
}

Python

import sys

n = int(sys.stdin.readline())
arr = [int(sys.stdin.readline()) for _ in range(n)]
sys.stdout.write("\n".join(map(str, arr)) + "\n")

5. Reading Large Input in One Go

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    string s;
    cin.ignore();
    getline(cin, s);
    cout << s << "\n";
    
    return 0;
}

Python

import sys

data = sys.stdin.read().split()
n = int(data[0])
arr = list(map(int, data[1:]))
sys.stdout.write(" ".join(map(str, arr)) + "\n")

6. Fast Output (Multiple Values)

C++

#include <iostream>
#include <sstream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    ostringstream out;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        out << x << " ";
    }
    
    cout << out.str() << "\n";
    return 0;
}

Python

import sys

n = int(sys.stdin.readline())
arr = list(map(int, sys.stdin.readline().split()))
sys.stdout.write(" ".join(map(str, arr)) + "\n")
