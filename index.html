In competitive programming, efficient input and output handling is crucial. Below is a concise and optimized approach for handling input and output in both C++ and Python.

1. Single Integer Input

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    cout << n << "\n";
    
    return 0;
}

Python

n = int(input())
print(n)

2. Multiple Space-Separated Inputs

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int a, b;
    cin >> a >> b;
    cout << a + b << "\n";
    
    return 0;
}

Python

a, b = map(int, input().split())
print(a + b)

3. Array Input (Space-Separated)

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << "\n";
    
    return 0;
}

Python

n = int(input())
arr = list(map(int, input().split()))
print(*arr)

4. Multiple Line Inputs

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    for (int i = 0; i < n; i++) {

    }
    
    return 0;
}

Python

n = int(input())
arr = [int(input()) for _ in range(n)]
print('\n'.join(map(str, arr)))

6. Fast Output (Multiple Values)

7. 2D Array Input

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n, m;
    cin >> n >> m;
    
    int arr[n][m];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> arr[i][j];
        }
    }
    
    // Output the 2D array
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << arr[i][j] << " ";
        }
        cout << "\n";
    }
    
    return 0;
}

Python

n = int(input())  # input number of test cases
for i in range(n):
m, n = map(int, input().split())  # input dimension for the 2D array
a = []
for j in range(m):
  a.append(list(map(int, input().split())))  # input the 2D array
for j in range(m):
  for k in range(n):
      print(a[j][k])  # print the 2D array

8. Multiple Test Cases

C++

#include <iostream>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        int arr[n];
        
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        
        // Process each test case
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            sum += arr[i];
        }
        
        cout << sum << "\n";
    }
    
    return 0;
}

Python

t = int(input())

for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))
    
    # Process each test case
    total = sum(arr)
    print(total)
